#打开文件夹，遍历文件夹内文件
if(is_dir($dirname)){
   $dir_handle=opendir($dirname);//打开文件夹
   while($file=readdir($dir_handle)){//读取文件或文件夹
     if($file!="."&&$file!=".."){
        $dir_file=$dirname."/".$file;
        if(filetype($dir_file) === 'file' && strpos($file, '.conf.php') !== false){
        	include_once $dir_file;
        }
     }
   }
   closedir($dir_handle);
}
