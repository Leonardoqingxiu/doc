static 方法里面不可以用this
引入一个类文件后，可以不实例化而直接调用，前提是此类文件中没有$this

#打开文件
if(is_dir($dirname)){
   $dir_handle=opendir($dirname);
   while($file=readdir($dir_handle)){
     if($file!="."&&$file!=".."){
        $dir_file=$dirname."/".$file;
        if(filetype($dir_file) === 'file' && strpos($file, '.conf.php') !== false){
        	include_once $dir_file;
        }
     }
   }
   closedir($dir_handle);
}
